53  color.red=map((0b00011100 & startcolor)>>2,0,7,0,255); //3 Bit green converted to 8 bit, assigned to red (color profiles!)
54  color.green=map((0b11100000 & startcolor)>>5,0,7,0,255);//3 Bit red converted to 8 bit, assigned to green (color profiles!)

58  //the converted values are assigned to the colors of the struct

62  color.red=map((0b11100000 & startcolor)>>5,0,7,0,255); //3 Bit red
63  }
64  return color;
